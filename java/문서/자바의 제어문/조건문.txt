* 조건문(=선택문)
   -if문, switch~case문

1. if문
  
   1)
    if(조건식){
         조건식이 참인경우 실행될 문장;
    }
   
   2)
   if(조건식){
         조건식이 참인경우 실행될 문장;
   }else{
         조건식이 거짓인경우 실행될 문장;
   }

   3)
   if(조건식1){
          조건식1이 참인경우 실행될 문장;
   }else if(조건식2){
          조건식2가 참인경우 실행될 문장;
   }else{
          위의 조건식을 만족하지 않을때 실행될 문장;
   }


2.  switch~case문

     switch(변수){

	case 값1: 실행될 문장;
		break;
	case 값2: 실행될 문장;
		break; 
	default: 위의 case를 만족하지 않을때
	            실행될 문장;
     }
   

// 여러개의 if-else if-else문을 사용하는 것 보다 가독성 높음
// 특정 값에 대한 다양한 동작 지정에 유리(단, 값에따라 분기하는 간단한 경우에만 적합)
switch (표현식) {	// 표현식은 조건을 평가하는데 사용됨
    // 표현식의 모든 값은 case의 값과 비교됨
    case 값1:
        // 코드 블록1
        break;
    case 값2:
        // 코드 블록2
        break;
    ...
    case 값n:
        // 코드 블록n
        break;
    default:	// 표현식의 값과 일치하는 case가 없을때 실행됨
        // 기본 코드 블록
}


#특정 조건에 따라서 수행해야 하는 경우 switch문의 default레이블 활용
-Variable부분 => 정수/문자/열거/문자열...(해당 자료형의 유효값으로 지정해야 함)
switch (variable) {
    case value1:
        // value1에 해당하는 동작 수행
        break;
    case value2:
        // value2에 해당하는 동작 수행
        break;
    default:	// 어떤 case레이블에도 해당하지 않는 경우 실행
        if (condition) {
            // 조건에 맞는 동작 수행
        } else {
            // 그 외의 경우에 해당하는 동작 수행
        }
}
